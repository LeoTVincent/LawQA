<script>

const csvURL="https://docs.google.com/spreadsheets/d/e/2PACX-1vQI_9xcBNvx2Pm38N5p0AvDp3zosdsRbgY8cQLUC6enncLXmvgpYx3J_aQupFKl-ne_q604KU8MgtBz/pub?output=csv";

let allCases=[];

Papa.parse(csvURL,{
download:true,
header:true,

complete:function(results){

allCases = results.data.filter(row =>
row["Party Name"] && row["Party Name"] !== "Party Name"
);

renderCases(allCases);

}

});

function safe(v){
return v ? v : "";
}

function highlight(text,keywords){

if(!keywords.length) return text;

keywords.forEach(k=>{
let regex=new RegExp(`(${k})`,"gi");
text=text.replace(regex,"<mark>$1</mark>");
});

return text;

}

function renderCases(data,query=""){

let keywords=query.trim().split(/\s+/).filter(Boolean);

let html="";

data.forEach(row=>{

html+=`

<div class="card">

<div class="case">
${safe(row["Serial Number"])} — ${highlight(safe(row["Party Name"]),keywords)}
</div>

<div class="section">
<span class="label">Court:</span>
${highlight(safe(row["06-Mar CCH"]),keywords)}
</div>

<div class="section">
<span class="label">Case:</span>
${highlight(safe(row["Case Type/Case Number/Case Year"]),keywords)}
</div>

<div class="section">
<span class="label">Advocate:</span>
${highlight(safe(row["Advocate"]),keywords)}
</div>

<div class="section">
<span class="label">Advocate For:</span>
${highlight(safe(row["Advocate for"]),keywords)}
</div>

<div class="section">
<span class="label">Stage:</span>
<span class="stage">${highlight(safe(row["Stage"]),keywords)}</span>
</div>

<div class="section">
<span class="label">Comments:</span>
${highlight(safe(row["Comments"]),keywords)}
</div>

</div>

`;

});

document.getElementById("cases").innerHTML=html;
document.getElementById("resultCount").innerText=data.length+" cases found";

}

function searchCase(){

let query=document.getElementById("searchBox").value.toLowerCase();

let keywords=query.trim().split(/\s+/).filter(Boolean);

let filtered=allCases.filter(row=>{

let text=Object.values(row).join(" ").toLowerCase();

return keywords.every(k=>text.includes(k));

});

renderCases(filtered,query);

}
