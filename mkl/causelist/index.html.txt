<script>

const csvURL="https://docs.google.com/spreadsheets/d/e/2PACX-1vQI_9xcBNvx2Pm38N5p0AvDp3zosdsRbgY8cQLUC6enncLXmvgpYx3J_aQupFKl-ne_q604KU8MgtBz/pub?output=csv";

let allCases=[];

Papa.parse(csvURL,{
download:true,
header:true,

complete:function(results){

allCases = results.data.filter(row =>
row["Party Name"] && row["Party Name"] !== "Party Name"
);

renderCases(allCases);

}

});

function safe(v){
return v ? v : "";
}

function highlight(text,keywords){

if(!keywords.length) return text;

keywords.forEach(k=>{
let regex=new RegExp(`(${k})`,"gi");
text=text.replace(regex,"<mark>$1</mark>");
});

return text;

}

function renderCases(data,query=""){

let keywords=query.trim().split(/\s+/).filter(Boolean);

let html="";

data.forEach(row=>{

html+=`

<tr>

<td>${safe(row["Serial Number"])}</td>

<td>${highlight(safe(row["06-Mar CCH"]),keywords)}</td>

<td>${highlight(safe(row["Case Type/Case Number/Case Year"]),keywords)}</td>

<td>${highlight(safe(row["Party Name"]),keywords)}</td>

<td>${highlight(safe(row["Advocate"]),keywords)}</td>

<td>${highlight(safe(row["Advocate for"]),keywords)}</td>

<td>${highlight(safe(row["Stage"]),keywords)}</td>

<td>${highlight(safe(row["Comments"]),keywords)}</td>

</tr>

`;

});

document.getElementById("cases").innerHTML=html;
document.getElementById("resultCount").innerText=data.length+" cases found";

}

function searchCase(){

let query=document.getElementById("searchBox").value.toLowerCase();

let keywords=query.trim().split(/\s+/).filter(Boolean);

let filtered=allCases.filter(row=>{

let text=Object.values(row).join(" ").toLowerCase();

return keywords.every(k=>text.includes(k));

});

renderCases(filtered,query);

}
